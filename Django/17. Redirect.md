```Python
def article_detail(request, article_id):
    article = get_object_or_404(Article, id=article_id)
    coms = Comment.objects.filter(is_show=True).filter(article_object_id=article_id)    
    if request.method == 'POST':
        form = CreateCommentForm(data=request.POST, files=request.FILES)
        if form.is_valid():
            comment_name = form.cleaned_data.get('name')
            comment_text = form.cleaned_data.get('text')
            comment_email = form.cleaned_data.get('email')
            comment_image = form.cleaned_data.get('image')
            comment_article = get_object_or_404(klass=Article, id=article_id)
            Comment.objects.create(name=comment_name, text=comment_text, email=comment_email, is_show=False, article_object=comment_article, image=comment_image)
            return redirect('blog:success_comment_name', name=comment_name)
            # name = comment_name: sending args to url for url parameters
    else:
        form = CreateCommentForm()
    context = {
        'article': article,
        'form': form,
        'coms': coms
    }

    return render(request, 'blog/detail_index.html', context)
```
Redirect helps to change user's URL.
